Kata "Convert string to camel case" at Codewars 
                            
function toCamelCase(str){
    let camelStr = '';
    let i = 0;
    while (i < str.length) {
        if (str[i] === '-' || str[i] === '_') {
            camelStr = camelStr + str[i+1].toUpperCase();
            i += 2;
        } else {
            camelStr = camelStr + str[i];
            i += 1;
        }
    }
    return camelStr;
}
